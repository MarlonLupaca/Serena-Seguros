-backup manual

docker exec -it mysql-backup bash -c "FECHA=`$(date +'%Y-%m-%d_%H-%M-%S_MANUAL') && mysqldump -h serena-mysql-replica -u root -proot bd_seguros_pacifico | gzip > /backups/backup_`$FECHA.sql.gz && echo 'Backup manual generado con éxito.'"
